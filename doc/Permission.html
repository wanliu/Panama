def self.define(resource, abilities)
  unless abilities.is_a?(Array)
    abilities = [abilities]
  end
  abilities.each do |ability|
    if find_by(resource: resource, ability: ability).nil?
      create(resource: resource, ability: ability)
    end
  end
end
