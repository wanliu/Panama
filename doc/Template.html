def initialize(name = nil, file_storage = nil)

    fs = file_storage || '/'.to_dir
    @fs = fs[@name]
  end

def ==(other)
  if other.equal?(self)
    return true
  elsif !other.instance_of?(self.class)
    return false
  end
  if name == other.name
    @fs.path == other.instance_variable_get(:@fs).path
  end
end
