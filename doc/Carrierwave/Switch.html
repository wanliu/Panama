def configure(&block)
  config = YAML::load_file("#{Rails.root}/config/switch_carrierwave.yml")
  storage = (config["upload_storage"] || "file").to_sym
  CarrierWave.configure{| config | config.storage = storage }
  yield(storage) if block_given?
end
