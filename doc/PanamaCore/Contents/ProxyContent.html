def compile_config
  result = {}
  [:root, :template].each do |method_name|
    _item = config[method_name]
    result[method_name] = _item.sub /\:(\w+)/ do |m|
                            name = m[1..-1].to_sym
                            locals[name]
                          end
  end
  result
end
