def value=(other)
  case property.property_type
  when /string/
    send(:svalue=, other)
  when /integer/
    send(:nvalue=, other)
  when /float/, /decimal/
    send(:dvalue=, other)
  when /datetime/
    send(:dtvalue=, other)
  when /set/
    item = valuable.property_items.select{|item| item.id.to_s == other}.first
    if item.present?
      send(:svalue=, item.value) #if valuable.property_items.select { |item| item.value == other }.size > 0
    end
  end
end
