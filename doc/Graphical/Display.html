def self.config(options = {})
  default_config = {
    :icon => "50x50",
    :header => "100x100",
    :avatar => "240x240",
    :preview => "420x420"
  }
  @config ||= default_config
  @config = default_config.merge(options) unless options.empty?
  @config[:default] = ""
  @config
end
