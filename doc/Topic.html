def as_json(*args)
  attribute = super *args
  attribute["user"] = {
    login: user.login,
    icon_url: user.photos.icon}
  attribute["category"] = {
    name: category_with_deleted.name
  }
  attribute["attachments"] = []
  attachments.each do |atta|
    attribute["attachments"] << atta.file.url
  end

  attribute
end
